42 DynamicModel modelFromString(const std::string& model) {

45 
46  std::map <std::string, ublox_gps::DynamicModel> models;

50  models["automotive"] = ublox_gps::DYN_MODEL_AUTOMOTIVE;
51  models["sea"] = ublox_gps::DYN_MODEL_SEA;
52  models["airborne1"] = ublox_gps::DYN_MODEL_AIRBORNE_1G;
53  models["airborne2"] = ublox_gps::DYN_MODEL_AIRBORNE_2G;
54  models["airborne4"] = ublox_gps::DYN_MODEL_AIRBORNE_4G;
55 
56  std::map<std::string, ublox_gps::DynamicModel>::iterator I = models.find(lower);
57  if (I == models.end()) {
58  throw std::runtime_error(lower + " is not a valid dynamic model.");
59  }
60  return I->second;
61 }

63 FixMode fixModeFromString(const std::string& mode) {
64  std::string lower = mode;

66  if (lower == "2d") {

239 void Gps::readCallback(unsigned char *data, std::size_t& size) {
240  ublox::Reader reader(data, size);
241 
242  while(reader.search() != reader.end() && reader.found()) {
243  if (debug >= 3) {
244  std::cout << "received ublox " << reader.length() + 8 << " bytes" << std::endl;
245  for(ublox::Reader::iterator it = reader.pos(); it != reader.pos() + reader.length() + 8; ++it) std::cout << std::hex << static_cast<unsigned int>(*it) << " ";
246  std::cout << std::dec << std::endl;
247  }
248 
249  callback_mutex_.lock();
250  Callbacks::key_type key = std::make_pair(reader.classId(), reader.messageId());
251  for(Callbacks::iterator callback = callbacks_.lower_bound(key); callback != callbacks_.upper_bound(key); ++callback) callback->second->handle(reader);
252  callback_mutex_.unlock();
253 
254  if (reader.classId() == 0x05) {
255  acknowledge_ = (reader.messageId() == 0x00) ? NACK : ACK;
256  if (debug >= 2) std::cout << "received " << (acknowledge_ == ACK ? "ACK" : "NACK") << std::endl;
257  }
258  }
259 
260  // delete read bytes from input buffer
261  std::copy(reader.pos(), reader.end(), data);
262  size -= reader.pos() - data;
263 }
