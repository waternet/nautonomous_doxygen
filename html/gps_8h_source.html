224 template <typename T> bool Gps::read(T& message, const boost::posix_time::time_duration& timeout) {
225  bool result = false;
226  if (!worker_) return false;
227 
228  callback_mutex_.lock();
229  CallbackHandler_<T> *handler = new CallbackHandler_<T>();
230  Callbacks::iterator callback = callbacks_.insert((std::make_pair(std::make_pair(T::CLASS_ID, T::MESSAGE_ID), boost::shared_ptr<CallbackHandler>(handler))));
231  callback_mutex_.unlock();
232 
233  if (handler->wait(timeout)) {
234  message = handler->get();
235  result = true;
236  }
237 
238  callback_mutex_.lock();
239  callbacks_.erase(callback);
240  callback_mutex_.unlock();
241  return result;
242 }

244 template <typename ConfigT> bool Gps::configure(const ConfigT& message, bool wait) {
245  if (!worker_) return false;
246 
247  acknowledge_ = WAIT;
248 
249  std::vector<unsigned char> out(1024);
250  ublox::Writer writer(out.data(), out.size());
251  if (!writer.write(message)) return false;
252  worker_->send(out.data(), writer.end() - out.data());
253 
254  if (!wait) return true;
255 
256  waitForAcknowledge(default_timeout_);
257  return (acknowledge_ == ACK);
