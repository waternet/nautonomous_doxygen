85 void publishNavVelNED(const ublox_msgs::NavVELNED& m)
86 {
87  static ros::Publisher publisher = nh->advertise<ublox_msgs::NavVELNED>("navvelned", kROSQueueSize);
88  publisher.publish(m);
89 
90  // Example geometry message
91  static ros::Publisher velocityPublisher =
92  nh->advertise<geometry_msgs::TwistWithCovarianceStamped>("fix_velocity", kROSQueueSize);
93  if (m.iTOW == last_nav_pos.iTOW) {
94  // use same time as las navposllh message
95  velocity.header.stamp = fix.header.stamp;
96  } else {
97  // create a new timestamp
98  velocity.header.stamp = ros::Time::now();
99  }
100  velocity.header.frame_id = frame_id;
101 
102  // convert to XYZ linear velocity
103  velocity.twist.twist.linear.x = m.velE/100.0;
104  velocity.twist.twist.linear.y = m.velN/100.0;
105  velocity.twist.twist.linear.z = -m.velD/100.0;
106 
107  const double stdSpeed = (m.sAcc/100.0) * 3;
108 
109  const int cols = 6;
110  velocity.twist.covariance[cols*0 + 0] = stdSpeed*stdSpeed;
111  velocity.twist.covariance[cols*1 + 1] = stdSpeed*stdSpeed;
112  velocity.twist.covariance[cols*2 + 2] = stdSpeed*stdSpeed;
113  velocity.twist.covariance[cols*3 + 3] = -1; // angular rate unsupported
114 
115  velocityPublisher.publish(velocity);
116  last_nav_vel = m;
117 }

119 void publishNavPosLLH(const ublox_msgs::NavPOSLLH& m)
120 {
121  static ros::Publisher publisher = nh->advertise<ublox_msgs::NavPOSLLH>("navposllh", kROSQueueSize);
122  publisher.publish(m);
123 
124  // Position message
125  static ros::Publisher fixPublisher = nh->advertise<sensor_msgs::NavSatFix>("fix", kROSQueueSize);
126  if (m.iTOW == last_nav_vel.iTOW) {
127  // use last timestamp
128  fix.header.stamp = velocity.header.stamp;
129  } else {
130  // new timestamp
131  fix.header.stamp = ros::Time::now();
132  }
133  fix.header.frame_id = frame_id;
134  fix.latitude = m.lat*1e-7;
135  fix.longitude = m.lon*1e-7;
136  fix.altitude = m.height*1e-3;
137  if (status.gpsFix >= status.GPS_2D_FIX)
138  fix.status.status = fix.status.STATUS_FIX;
139  else
140  fix.status.status = fix.status.STATUS_NO_FIX;
141 
142  // calculate covariance (convert from mm to m too)
143  const double stdH = (m.hAcc / 1000.0) * 3.0;
144  const double stdV = (m.vAcc / 1000.0) * 3.0;
145 
146  fix.position_covariance[0] = stdH*stdH;
147  fix.position_covariance[4] = stdH*stdH;
148  fix.position_covariance[8] = stdV*stdV;
149  fix.position_covariance_type =
150  sensor_msgs::NavSatFix::COVARIANCE_TYPE_DIAGONAL_KNOWN;
151 
152  fix.status.service = fix.status.SERVICE_GPS;
153  fixPublisher.publish(fix);
154  last_nav_pos = m;
155  // update diagnostics
156  freq_diag->tick(fix.header.stamp);
157  updater->update();
158 }

238 void fix_diagnostic(diagnostic_updater::DiagnosticStatusWrapper &stat) {
239  // check the last message, convert to diagnostic
240  if (status.gpsFix == ublox_msgs::NavSTATUS::GPS_NO_FIX) {
241  stat.level = diagnostic_msgs::DiagnosticStatus::ERROR;
242  stat.message = "No fix";
243  }
244  else if (status.gpsFix == ublox_msgs::NavSTATUS::GPS_DEAD_RECKONING_ONLY) {
245  stat.level = diagnostic_msgs::DiagnosticStatus::WARN;
246  stat.message = "Dead reckoning only";
247  }
248  else if (status.gpsFix == ublox_msgs::NavSTATUS::GPS_2D_FIX) {
249  stat.level = diagnostic_msgs::DiagnosticStatus::OK;
250  stat.message = "2D fix";
251  }
252  else if (status.gpsFix == ublox_msgs::NavSTATUS::GPS_3D_FIX) {
253  stat.level = diagnostic_msgs::DiagnosticStatus::OK;
254  stat.message = "3D fix";
255  }
256  else if (status.gpsFix == ublox_msgs::NavSTATUS::GPS_GPS_DEAD_RECKONING_COMBINED) {
257  stat.level = diagnostic_msgs::DiagnosticStatus::OK;
258  stat.message = "GPS and dead reckoning combined";
259  }
260  else if (status.gpsFix == ublox_msgs::NavSTATUS::GPS_TIME_ONLY_FIX) {
261  stat.level = diagnostic_msgs::DiagnosticStatus::WARN;
262  stat.message = "Time fix only";
263  }
264 
265  // append last fix position
266  stat.add("iTOW", last_nav_pos.iTOW);
267  stat.add("lon", last_nav_pos.lon);
268  stat.add("lat", last_nav_pos.lat);
269  stat.add("height", last_nav_pos.height);
270  stat.add("hMSL", last_nav_pos.hMSL);
271  stat.add("hAcc", last_nav_pos.hAcc);
272  stat.add("vAcc", last_nav_pos.vAcc);
273  stat.add("numSV", num_svs_used);
274 }

276 int main(int argc, char **argv) {
277  boost::asio::io_service io_service;
278  ros::Timer poller;
279  boost::shared_ptr<boost::asio::ip::tcp::socket> tcp_handle;
280  boost::shared_ptr<boost::asio::serial_port> serial_handle;
281  bool setup_ok = true;
282 
283  ros::init(argc, argv, "ublox_gps");
284  nh.reset(new ros::NodeHandle("~"));
285  if (!nh->hasParam("diagnostic_period")) {
286  nh->setParam("diagnostic_period", 0.2); // 5Hz diagnostic period
287  }
288  updater.reset(new diagnostic_updater::Updater());
289  updater->setHardwareID("ublox");
290 
291  std::string device;
292  int baudrate;
293  int rate, meas_rate;
294  bool enable_sbas, enable_glonass, enable_beidou, enable_ppp;
295  std::string dynamic_model, fix_mode;
296  int dr_limit;
297  int ublox_version;
298  ros::NodeHandle param("~");
299  param.param("device", device, std::string("/dev/nautonomous/gps"));
300  param.param("frame_id", frame_id, std::string("gps"));
301  param.param("baudrate", baudrate, 9600);
302  param.param("rate", rate, 4); // in Hz
303  param.param("enable_sbas", enable_sbas, false);
304  param.param("enable_glonass", enable_glonass, false);
305  param.param("enable_beidou", enable_beidou, false);
306  param.param("enable_ppp", enable_ppp, false);
307  param.param("dynamic_model", dynamic_model, std::string("portable"));
308  param.param("fix_mode", fix_mode, std::string("both"));
309  param.param("dr_limit", dr_limit, 0);
310  param.param("ublox_version", ublox_version, 6);
311 
312  if (enable_ppp) {
313  ROS_WARN("Warning: PPP is enabled - this is an expert setting.");
314  }
315 
316  if (rate <= 0) {
317  ROS_ERROR("Invalid settings: rate must be > 0");
318  return 1;
319  }
320  // measurement rate param for ublox, units of ms
321  meas_rate = 1000 / rate;
322 
323  if (dr_limit < 0 || dr_limit > 255) {
324  ROS_ERROR("Invalid settings: dr_limit must be between 0 and 255");
325  return 1;
326  }
327 
328  DynamicModel dmodel;
329  FixMode fmode;
330  try {
331  dmodel = ublox_gps::modelFromString(dynamic_model);
332  fmode = ublox_gps::fixModeFromString(fix_mode);
333  } catch (std::exception& e) {
334  ROS_ERROR("Invalid settings: %s", e.what());
335  return 1;
336  }
337 
338  // configure diagnostic updater for frequency
339  updater->add("fix", &fix_diagnostic);
340  updater->force_update();
341 
342  const double target_freq = 1000.0 / meas_rate; // actual update frequency
343  double min_freq = target_freq;
344  double max_freq = target_freq;
345  diagnostic_updater::FrequencyStatusParam freq_param(&min_freq, &max_freq, 0.05, 10);
346  diagnostic_updater::TimeStampStatusParam time_param(0,meas_rate * 1e-3 * 0.05);
347  freq_diag.reset(new diagnostic_updater::TopicDiagnostic(std::string("fix"),
348  *updater,
349  freq_param,
350  time_param));
351 
352  boost::smatch match;
353  if (boost::regex_match(device, match, boost::regex("(tcp|udp)://(.+):(\\d+)"))) {
354  std::string proto(match[1]);
355  std::string host(match[2]);
356  std::string port(match[3]);
357  ROS_INFO("Connecting to %s://%s:%s ...", proto.c_str(), host.c_str(), port.c_str());
358 
359  if (proto == "tcp") {
360  boost::asio::ip::tcp::resolver::iterator endpoint;
361 
362  try {
363  boost::asio::ip::tcp::resolver resolver(io_service);
364  endpoint = resolver.resolve(boost::asio::ip::tcp::resolver::query(host, port));
365  } catch(std::runtime_error& e) {
366  ROS_ERROR("Could not resolve %s:%s: %s", host.c_str(), port.c_str(), e.what());
367  return 1; // exit
368  }
369 
370  boost::asio::ip::tcp::socket *socket = new boost::asio::ip::tcp::socket(io_service);
371  tcp_handle.reset(socket);
372 
373  try {
374  socket->connect(*endpoint);
375  } catch(std::runtime_error& e) {
376  ROS_ERROR("Could not connect to %s:%s: %s", endpoint->host_name().c_str(), endpoint->service_name().c_str(), e.what());
377  return 1; // exit
378  }
379 
380  ROS_INFO("Connected to %s:%s.", endpoint->host_name().c_str(), endpoint->service_name().c_str());
381  gps.initialize(*socket, io_service);
382  } else {
383  ROS_ERROR("Protocol '%s' is unsupported", proto.c_str());
384  return 1; // exit
385  }
386  } else {
387  boost::asio::serial_port *serial = new boost::asio::serial_port(io_service);
388  serial_handle.reset(serial);
389 
390  // open serial port
391  try {
392  serial->open(device);
393  } catch (std::runtime_error& e) {
394  ROS_ERROR("Could not open serial port %s: %s", device.c_str(), e.what());
395  return 1; // exit
396  }
397 
398  ROS_INFO("Opened serial port %s", device.c_str());
399  gps.setBaudrate(baudrate);
400  gps.initialize(*serial, io_service);
401  }
402 
403  // apply all requested settings
404  try {
405  if (!gps.isInitialized()) {
406  throw std::runtime_error("Failed to initialize.");
407  }
408  if (!gps.setMeasRate(meas_rate)) {
409  std::stringstream ss;
410  ss << "Failed to set measurement rate to " << meas_rate << "ms.";
411  throw std::runtime_error(ss.str());
412  }
413  if (!gps.enableSBAS(enable_sbas)) {
414  throw std::runtime_error(std::string("Failed to ") +
415  ((enable_sbas) ? "enable" : "disable") + " SBAS.");
416  }
417  if (!gps.setPPPEnabled(enable_ppp)) {
418  throw std::runtime_error(std::string("Failed to ") +
419  ((enable_ppp) ? "enable" : "disable") + " PPP.");
420  }
421  if (!gps.setDynamicModel(dmodel)) {
422  throw std::runtime_error("Failed to set model: " +
423  dynamic_model + ".");
424  }
425  if (!gps.setFixMode(fmode)) {
426  throw std::runtime_error("Failed to set fix mode: " + fix_mode + ".");
427  }
428  if (!gps.setDeadReckonLimit(dr_limit)) {
429  std::stringstream ss;
430  ss << "Failed to set dead reckoning limit: " << dr_limit << ".";
431  throw std::runtime_error(ss.str());
432  }
433 
434  if (ublox_version >= 7) {
435  ublox_msgs::CfgGNSS cfgGNSS;
436  if (gps.poll(cfgGNSS)) {
437  ROS_INFO("Read GNSS config.");
438  ROS_INFO("Num. tracking channels in hardware: %i",cfgGNSS.numTrkChHw);
439  ROS_INFO("Num. tracking channels to use: %i",cfgGNSS.numTrkChUse);
440  } else {
441  throw std::runtime_error("Failed to read the GNSS config.");
442  }
443 
444  cfgGNSS.numConfigBlocks = 1; // do services one by one
445  cfgGNSS.msgVer = 0;
446  cfgGNSS.resTrkCh = 8; // taken as defaults from ublox manual
447  cfgGNSS.maxTrkCh = 16;
448 
449  // configure glonass
450  cfgGNSS.gnssId = ublox_msgs::CfgGNSS::GNSS_ID_GLONASS;
451  cfgGNSS.flags = enable_glonass;
452  if (!gps.configure(cfgGNSS)) {
453  throw std::runtime_error(std::string("Failed to ") +
454  ((enable_glonass) ? "enable" : "disable") + " GLONASS.");
455  }
456  if (ublox_version >= 8) {
457  // configure beidou
458  cfgGNSS.gnssId = ublox_msgs::CfgGNSS::GNSS_ID_BEIDOU;
459  cfgGNSS.flags = enable_beidou;
460  if (!gps.configure(cfgGNSS)) {
461  throw std::runtime_error(std::string("Failed to ") +
462  ((enable_beidou) ? "enable" : "disable") + " BeiDou.");
463  }
464  } else {
465  ROS_WARN("ublox_version < 8, ignoring BeiDou Settings");
466  }
467  } else {
468  ROS_WARN("ublox_version < 7, ignoring GNSS settings");
469  }
470  } catch (std::exception& e) {
471  setup_ok = false;
472  ROS_ERROR("Error configuring device: %s", e.what());
473  }
474 
475  if (setup_ok) {
476  ROS_INFO("U-Blox configured successfully.");
477 
478  // subscribe messages
479  param.param("all", enabled["all"], false);
480  param.param("rxm", enabled["rxm"], false);
481  param.param("aid", enabled["aid"], false);
482 
483  param.param("nav_sol", enabled["nav_sol"], true);
484  if (enabled["nav_sol"]) gps.subscribe<ublox_msgs::NavSOL>(boost::bind(&publish<ublox_msgs::NavSOL>, _1, "navsol"), 1);
485  param.param("nav_status", enabled["nav_status"], true);
486  if (enabled["nav_status"]) gps.subscribe<ublox_msgs::NavSTATUS>(&publishNavStatus, 1);
487  param.param("nav_svinfo", enabled["nav_svinfo"], enabled["all"]);
488  if (enabled["nav_svinfo"]) gps.subscribe<ublox_msgs::NavSVINFO>(&publishNavSVINFO, 20);
489  param.param("nav_clk", enabled["nav_clk"], enabled["all"]);
490  if (enabled["nav_clk"]) gps.subscribe<ublox_msgs::NavCLOCK>(&publishNavCLK, 1);
491  param.param("rxm_raw", enabled["rxm_raw"], enabled["all"] || enabled["rxm"]);
492  if (enabled["rxm_raw"]) gps.subscribe<ublox_msgs::RxmRAW>(&publishRxmRAW, 1);
493  param.param("rxm_sfrb", enabled["rxm_sfrb"], enabled["all"] || enabled["rxm"]);
494  if (enabled["rxm_sfrb"]) gps.subscribe<ublox_msgs::RxmSFRB>(&publishRxmSFRB, 1);
495  param.param("nav_posllh", enabled["nav_posllh"], true);
496  if (enabled["nav_posllh"]) gps.subscribe<ublox_msgs::NavPOSLLH>(&publishNavPosLLH, 1);
497  param.param("nav_velned", enabled["nav_velned"], true);
498  if (enabled["nav_velned"]) gps.subscribe<ublox_msgs::NavVELNED>(&publishNavVelNED, 1);
499  param.param("aid_alm", enabled["aid_alm"], enabled["all"] || enabled["aid"]);
500  if (enabled["aid_alm"]) gps.subscribe<ublox_msgs::AidALM>(&publishAidALM);
501  param.param("aid_eph", enabled["aid_eph"], enabled["all"] || enabled["aid"]);
502  if (enabled["aid_eph"]) gps.subscribe<ublox_msgs::AidEPH>(&publishAidEPH);
503  param.param("aid_hui", enabled["aid_hui"], enabled["all"] || enabled["aid"]);
504  if (enabled["aid_hui"]) gps.subscribe<ublox_msgs::AidHUI>(&publishAidHUI);
505 
506  poller = nh->createTimer(ros::Duration(1.0), &pollMessages);
507  poller.start();
508  ros::spin();
509  }
510 
511  if (gps.isInitialized()) {
512  gps.close();
513  ROS_INFO("Closed connection to %s.", device.c_str());
514  }
515  return 0;
516 }
