203 Serial::readlines (size_t size, string eol)

208  uint8_t *buffer_ = static_cast<uint8_t*>

210  size_t read_so_far = 0;
211  size_t start_of_line = 0;
212  while (read_so_far < size) {
213  size_t bytes_read = this->read_ (buffer_+read_so_far, 1);
214  read_so_far += bytes_read;
215  if (bytes_read == 0) {
216  if (start_of_line != read_so_far) {
217  lines.push_back (
218  string (reinterpret_cast<const char*> (buffer_ + start_of_line),
219  read_so_far - start_of_line));
220  }
221  break; // Timeout occured on reading 1 byte
222  }
223  if (string (reinterpret_cast<const char*>

225  // EOL found
226  lines.push_back(
227  string(reinterpret_cast<const char*> (buffer_ + start_of_line),
228  read_so_far - start_of_line));
